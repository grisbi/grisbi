00046 gdouble bet_data_finance_get_echeance ( gdouble capital,
00047                         gdouble taux_periodique,
00048                         gint nbre_echeances )
00049 {
00050     gdouble coeff;
00051     gdouble number;
00052 
00053     coeff = pow ( ( 1 + taux_periodique ), nbre_echeances );
00054 
00055     if ( taux_periodique == 0 )
00056         number = capital / nbre_echeances;
00057     else
00058         number = ( capital * taux_periodique * coeff ) / ( coeff - 1 );
00059     number = bet_data_finance_troncate_number ( number, 2 );
00060 
00061     return number;
00062 }
