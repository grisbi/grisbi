00502 gint gsb_form_widget_next_element ( gint account_number,
00503                         gint element_number,
00504                         gint direction )
00505 {
00506     gint row;
00507     gint column;
00508     gint return_value_number = 0;
00509     gint form_column_number;
00510     gint form_row_number;
00511     
00512     if ( !gsb_data_form_look_for_value ( account_number,
00513                                          element_number,
00514                                          &row,
00515                                          &column ))
00516         return -1;
00517 
00518     form_column_number = gsb_data_form_get_nb_columns (account_number);
00519     form_row_number = gsb_data_form_get_nb_rows (account_number);
00520 
00521     while ( !gsb_form_widget_can_focus (return_value_number)) 
00522     {
00523         switch ( direction )
00524         {
00525             case GSB_LEFT:
00526                 if ( !column && !row )
00527                 {
00528                     /* we are at the upper left, go on the bottom right */
00529                     column = form_column_number;
00530                     row = form_row_number -1; 
00531                 }
00532 
00533                 if ( --column == -1 )
00534                 {
00535                     column = form_column_number - 1;
00536                     row--;
00537                 }
00538                 return_value_number = gsb_data_form_get_value ( account_number,
00539                                                                 column,
00540                                                                 row );
00541                 break;
00542 
00543             case GSB_RIGHT:
00544                 if ( column == (form_column_number - 1)
00545                      &&
00546                      row == (form_row_number - 1))
00547                 {
00548                     /* we are on the bottom right, we finish the edition or
00549                      * go to the upper left */
00550                     if ( !conf.entree )
00551                     {
00552                         return_value_number = -2;
00553                         continue;
00554                     }
00555                     column = -1;
00556                     row = 0; 
00557                 }
00558 
00559                 if ( ++column == form_column_number )
00560                 {
00561                     column = 0;
00562                     row++;
00563                 }
00564                 return_value_number = gsb_data_form_get_value ( account_number,
00565                                                                 column,
00566                                                                 row );
00567                 break;
00568 
00569             case GSB_UP:
00570                 if ( !row )
00571                 {
00572                     return_value_number = -1;
00573                     continue;
00574                 }
00575 
00576                 row--;
00577                 return_value_number = gsb_data_form_get_value ( account_number,
00578                                                                 column,
00579                                                                 row );
00580                 break;
00581 
00582             case GSB_DOWN:
00583                 if ( row == (form_row_number - 1))
00584                 {
00585                     return_value_number = -1;
00586                     continue;
00587                 }
00588                 row++;
00589                 return_value_number = gsb_data_form_get_value ( account_number,
00590                                                                 column,
00591                                                                 row );
00592                 break;
00593 
00594             default:
00595                 return_value_number = -1;
00596         }
00597     }
00598     return return_value_number;
00599 }

00609 gboolean gsb_form_widget_can_focus ( gint element_number )
00610 {
00611     GtkWidget *widget;
00612 
00613     /* if element_number is -1 or -2, the iteration while must
00614      * stop, so return TRUE */
00615     if ( element_number == -1
00616          ||
00617          element_number == -2 )
00618         return TRUE;
00619 
00620     widget = gsb_form_widget_get_widget (element_number);
00621 
00622     if ( !widget )
00623         return FALSE;
00624 
00625     if ( !GTK_WIDGET_VISIBLE (widget))
00626         return FALSE;
00627 
00628     if ( !GTK_WIDGET_SENSITIVE (widget))
00629         return FALSE;
00630 
00631     if ( !(GTK_IS_COMBOFIX (widget)
00632            ||
00633            GTK_IS_ENTRY (widget)
00634            ||
00635            GTK_IS_BUTTON (widget)
00636            ||
00637            GTK_IS_COMBO_BOX (widget)))
00638         return FALSE;
00639 
00640     return TRUE;

00960 gboolean gsb_form_widget_get_valide_amout_entry ( const gchar *string )
00961 {
00962     struct lconv *conv = localeconv ( );
00963     gchar *mon_thousands_sep;
00964     gunichar thousands_sep;
00965     const gchar *ptr;
00966 
00967     ptr = string;
00968     mon_thousands_sep = g_locale_to_utf8 ( conv->mon_thousands_sep,
00969                         -1, NULL, NULL, NULL );
00970     thousands_sep = g_utf8_get_char_validated ( mon_thousands_sep, -1 );
00971     
00972     while ( g_utf8_strlen (ptr, -1) > 0 )
00973     {
00974         gunichar ch;
00975 
00976         ch = g_utf8_get_char_validated (ptr, -1);
00977         if ( !g_unichar_isdefined ( ch ) )
00978             return FALSE;
00979         if ( !g_ascii_isdigit ( ch ) )
00980         {
00981             if ( g_unichar_isdefined ( thousands_sep ) )
00982             {
00983                 if ( ch != '.' && ch != ',' && ch != '+' && ch != '-'
00984                  && ch != thousands_sep
00985                   )
00986                     return FALSE;
00987             }
00988             else if ( ch != '.' && ch != ',' && ch != '+' && ch != '-' )
00989                     return FALSE;
00990         }
00991 
00992         ptr = g_utf8_next_char (ptr);
00993     }
00994 
00995     return TRUE;
00996 }
