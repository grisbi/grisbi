00231 gboolean gsb_calendar_entry_key_press ( GtkWidget *entry,
00232                                         GdkEventKey  *ev,
00233                                         gpointer null )
00234 {
00235     switch ( ev -> keyval )
00236     {
00237         case GDK_KP_Enter :
00238         case GDK_Return :
00239 
00240             /* CONTROL ENTER opens the calendar */
00241             if ( ( ev -> state & GDK_CONTROL_MASK ) == GDK_CONTROL_MASK)
00242                 gsb_calendar_entry_popup (entry);
00243             break;
00244 
00245         case GDK_KP_Add:
00246         case GDK_plus:
00247         case GDK_equal:         /* This should make all our US users happy */
00248 
00249             /* increase the date of 1 day/week */
00250             if ( ( ev -> state & GDK_CONTROL_MASK ) != GDK_CONTROL_MASK ||
00251                  ev -> keyval != GDK_KP_Add )
00252                 gsb_calendar_entry_step_date ( entry, ONE_DAY );
00253             else
00254                 gsb_calendar_entry_step_date ( entry, ONE_WEEK );
00255             return TRUE;
00256             break;
00257 
00258         case GDK_KP_Subtract:
00259         case GDK_minus:
00260 
00261             /* decrease the date of 1 day/week, or the check of 1 */
00262             if ( ( ev -> state & GDK_CONTROL_MASK ) != GDK_CONTROL_MASK ||
00263                  ev -> keyval != GDK_KP_Subtract  )
00264                 gsb_calendar_entry_step_date ( entry, -ONE_DAY );
00265             else
00266                 gsb_calendar_entry_step_date ( entry, -ONE_WEEK );
00267             return TRUE;
00268             break;
00269 
00270         case GDK_Page_Up :
00271         case GDK_KP_Page_Up :
00272 
00273             /* increase the date of 1 month/year */
00274             if ( ( ev -> state & GDK_CONTROL_MASK ) != GDK_CONTROL_MASK )
00275                 gsb_calendar_entry_step_date ( entry,
00276                                ONE_MONTH );

00279                                ONE_YEAR );
00280 
00281             return TRUE;
00282             break;
00283 
00284         case GDK_Page_Down :
00285         case GDK_KP_Page_Down :
00286 
00287             /* decrease the date of 1 month/year */

00350             break ;

00400                                       gpointer null )
00401 {
00402     GDate *date;
00403 
00404     /* if we are in the form and the entry is empty, do nothing
00405      * because it's a special style too */
00406     if (gsb_form_widget_check_empty (entry))

00409     /* if nothing in the entry, keep the normal style */
00410     if (!strlen (gtk_entry_get_text (GTK_ENTRY (entry))))
00411     {
00412         gsb_calendar_entry_set_color ( entry, TRUE );
00413         return FALSE;

00415 
00416     /* to check the date, we just try to see if can have a dote from the entry */
00417     date = gsb_date_get_last_entry_date (gtk_entry_get_text (GTK_ENTRY (entry)));
00418 
00419     if (date)
00420     {
00421         /* the date is valid, make it normal */
00422         gsb_calendar_entry_set_color ( entry, TRUE );
00423         g_date_free (date);
00424     }
00425     else
00426     {
00427         /* the date is not valid, make it red */
00428         gsb_calendar_entry_set_color ( entry, FALSE );
00429     }
00430

00580 gboolean gsb_calendar_entry_calendar_key_press ( GtkCalendar *pCalendar,
00581                                                  GdkEventKey *ev,
00582                                                  GtkWidget *entry )
00583 {
00584     guint day, month, year;
00585     GDate *date;
00586     GtkWidget *pTopLevelWidget;
00587 
00588     /* get the popup to destroy it if need */
00589     pTopLevelWidget = gtk_widget_get_toplevel ( GTK_WIDGET ( pCalendar ) );
00590 
00591     /* most of the time, we will use date so can get it here,
00592      * think about free it if not used */
00593     gtk_calendar_get_date ( pCalendar, &year, &month, &day );
00594     month++;
00595     date = g_date_new_dmy (day, month, year);
00596 
00597     switch ( ev -> keyval )
00598     {
00599         case GDK_Escape :
00600             /* just close the calendar if it's a popup */
00601             if ( GTK_WIDGET_TOPLEVEL ( pTopLevelWidget ) )
00602                 gtk_widget_destroy ( pTopLevelWidget );
00603             g_date_free (date);
00604             return TRUE;
00605             break ;
00606 
00607         case GDK_Return :
00608         case GDK_KP_Enter :
00609             /* get the date an close the calendar */
00610             gtk_entry_set_text ( GTK_ENTRY ( entry ),
00611                                  gsb_format_date ( day, month + 1, year ));
00612             if ( GTK_WIDGET_TOPLEVEL ( pTopLevelWidget ) )
00613                 gtk_widget_destroy ( pTopLevelWidget );
00614             g_date_free (date);
00615             return TRUE;
00616             break ;
00617 
00618             /* from now, it will change date so just use date, modify it and fill day, month, year
00619              * we will set the calendar at the end of that function
00620              * so after now, only keys which change the date */
00621         case GDK_Left :
00622         case GDK_KP_Left:
00623         case GDK_minus:
00624         case GDK_KP_Subtract:
00625             /* day before */
00626             g_date_subtract_days (date, 1);
00627             break ;
00628 
00629         case GDK_Right :
00630         case GDK_KP_Right:
00631         case GDK_plus:
00632         case GDK_KP_Add:
00633             /* day after */
00634             g_date_add_days (date, 1);
00635             break ;
00636 
00637         case GDK_Up :
00638         case GDK_KP_Up :
00639             /* prev week */
00640             g_date_subtract_days (date, 7);
00641             break ;
00642 
00643         case GDK_Down :
00644         case GDK_KP_Down :
00645             /* next week */
00646             g_date_add_days (date, 7);
00647             break ;
00648 
00649         case GDK_Home :
00650         case GDK_KP_Home :
00651             /* go to first day of the month */
00652             g_date_set_day (date, 1);
00653             break ;
00654 
00655         case GDK_End :
00656         case GDK_KP_End :
00657             /* go to last day of the month */
00658             g_date_set_day (date,

00660             break ;
00661 
00662         case GDK_Page_Up :
00663         case GDK_KP_Page_Up :
00664             /* prev month */

00666             break ;
00667

00669         case GDK_KP_Page_Down :

00678     day = g_date_get_day (date);
