00393 void go_option_menu_set_history (GOOptionMenu *option_menu, GSList *selection)
00394 {
00395   GtkWidget *item;
00396   
00397   g_return_if_fail (selection != NULL);
00398   g_return_if_fail (GO_IS_OPTION_MENU (option_menu));
00399 
00400   if (option_menu->menu)
00401     {
00402             GtkMenu *menu = GTK_MENU(option_menu->menu);
00403             
00404             while (selection->next) {
00405                     GList *children = gtk_container_get_children (GTK_CONTAINER(menu));
00406                     gint index = GPOINTER_TO_INT (selection->data);
00407                     item = GTK_WIDGET(g_list_nth_data (children, index));
00408                     menu = GTK_MENU(gtk_menu_item_get_submenu (GTK_MENU_ITEM(item)));
00409                     selection = selection->next;
00410                     g_list_free (children);
00411             }
00412             
00413             gtk_menu_set_active (menu,  GPOINTER_TO_INT (selection->data));
00414             item = gtk_menu_get_active (menu);
00415             if (item != option_menu->menu_item)
00416                     go_option_menu_update_contents (option_menu, menu);
00417             g_slist_free (option_menu->selection);
00418             option_menu->selection = g_slist_copy (selection);
00419     }
00420 }

00529   GOOptionMenu *option_menu = GO_OPTION_MENU (widget);
00530   GOOptionMenuProps props;
00531   gint tmp;
00532   GtkRequisition child_requisition = { 0, 0 };

00535  
00536   if (GTK_BIN (option_menu)->child && GTK_WIDGET_VISIBLE (GTK_BIN (option_menu)->child))
00537     {
00538       gtk_widget_size_request (GTK_BIN (option_menu)->child, &child_requisition);
00539 
00540       requisition->width += child_requisition.width;
00541       requisition->height += child_requisition.height;
00542     }
00543   
00544   requisition->width = ((GTK_CONTAINER (widget)->border_width +
00545                          GTK_WIDGET (widget)->style->xthickness + props.focus_pad) * 2 +
00546                         MAX (child_requisition.width, option_menu->width) +
00547                         props.indicator_size.width +
00548                         props.indicator_spacing.left + props.indicator_spacing.right +
00549                         CHILD_LEFT_SPACING + CHILD_RIGHT_SPACING + props.focus_width * 2);
00550   requisition->height = ((GTK_CONTAINER (widget)->border_width +
00551                           GTK_WIDGET (widget)->style->ythickness + props.focus_pad) * 2 +
00552                          MAX (child_requisition.height, option_menu->height) +
00553                          CHILD_TOP_SPACING + CHILD_BOTTOM_SPACING + props.focus_width * 2);
00554 
00555   tmp = (requisition->height - MAX (child_requisition.height, option_menu->height) +
00556          props.indicator_size.height + props.indicator_spacing.top + props.indicator_spacing.bottom);
00557   requisition->height = MAX (requisition->height, tmp);
